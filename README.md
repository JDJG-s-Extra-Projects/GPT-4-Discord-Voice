# Install Notes
debian:  
    sudo apt install ffmpeg  
    pip install elevenlabs  
    pip install elevenlabslib  
    sudo apt-get install libportaudio2  
    sudo apt-get install python3-pyaudio  
    python3 -m pip install -U discord.py  
    pip install gTTS
